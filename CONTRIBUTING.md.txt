name: CI
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Run tests
        run: cargo test --all
  fmt:
    runs-on: ubuntu-latest

      - uses: actions/checkout@v4
      - name: Check formatting
        run: cargo fmt -- --check
